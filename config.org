* Package Management
Bootstrapping di `use-package` e rimozione della schermata di avvio.

#+begin_src emacs-lisp
  (setq inhibit-startup-message t
        initial-scratch-message ""
        initial-buffer-choice t)

  (unless (package-installed-p 'use-package)
    (package-refresh-contents)
    (package-install 'use-package))
  (require 'use-package)
  (setq use-package-always-ensure t)
#+end_src

* Minibuffer e Navigazione
Framework di ricerca, completamento e gestione buffer.

#+begin_src emacs-lisp
  (use-package vertico
    :init (vertico-mode 1)
    :custom (vertico-cycle t))

  (use-package savehist
    :ensure nil
    :init (savehist-mode 1))

  (use-package marginalia
    :init (marginalia-mode 1))

  (use-package orderless
    :custom
    (completion-styles '(orderless basic))
    (completion-category-defaults nil)
    (completion-category-overrides '((file (styles partial-completion)))))

  (use-package consult
    :bind (("C-x b" . consult-buffer)
           ("C-s"   . consult-line)
           ("C-c s" . consult-ripgrep))
    :custom
    (isearch-invisible 'open)
    :config
    (require 'cl-lib)
    
    (defmacro my/consult-source (name key &rest conds)
      `(list :name ,name :narrow ,key :category 'buffer :face 'consult-buffer
             :history 'buffer-name-history :state #'consult--buffer-state
             :items (lambda () (mapcar #'buffer-name
               (cl-remove-if-not (lambda (b) (with-current-buffer b (and (not (string-prefix-p " " (buffer-name b))) ,@conds)))
                                 (buffer-list))))))

    ;; Utilizzo defvar per inizializzazione sicura
    (defvar my/consult-source-vterm  (my/consult-source "Vterm"  ?v (derived-mode-p 'vterm-mode)))
    (defvar my/consult-source-org    (my/consult-source "Org"    ?o (derived-mode-p 'org-mode)))
    (defvar my/consult-source-buffer (my/consult-source "Buffer" ?b (not (or (derived-mode-p 'vterm-mode) (derived-mode-p 'org-mode)))))

    (setq consult-buffer-sources (delq 'consult--source-buffer consult-buffer-sources))
    (add-to-list 'consult-buffer-sources 'my/consult-source-buffer t)
    (add-to-list 'consult-buffer-sources 'my/consult-source-org t)
    (add-to-list 'consult-buffer-sources 'my/consult-source-vterm t))

  (use-package which-key
    :init (which-key-mode)
    :custom (which-key-idle-delay 0.5))
#+end_src

* Terminale (Vterm)
Emulatori, tiling e helper generali dei buffer.

#+begin_src emacs-lisp
  (use-package vterm
    :commands vterm
    :custom (vterm-max-scrollback 10000))

  (use-package vterm-toggle
    :bind (("C-c t" . vterm-toggle)
           ("C-c T" . vterm-toggle-cd))
    :custom (vterm-toggle-fullscreen-p nil)
    :config
    (add-to-list 'display-buffer-alist
                 '((lambda (buf-name _) (with-current-buffer buf-name (eq major-mode 'vterm-mode)))
                    (display-buffer-reuse-window display-buffer-at-bottom)
                    (reusable-frames . visible)
                    (window-height . 0.3))))

  (use-package multi-vterm
    :bind (("C-c v c" . multi-vterm)
           ("C-c v 3" . my/vterm-split-right)
           ("C-c v 2" . my/vterm-split-below))
    :config
    (defun my/vterm-split-right ()
      (interactive) (split-window-right) (other-window 1) (multi-vterm))
    (defun my/vterm-split-below ()
      (interactive) (split-window-below) (other-window 1) (multi-vterm)))

  ;; Utility per rinominare sessioni attive al volo
  (global-set-key (kbd "C-c r") 'rename-buffer)
#+end_src
