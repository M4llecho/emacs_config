* Package Management
Bootstrapping di =use-package= e pin di sicurezza per i pacchetti built-in.

#+begin_src emacs-lisp
  (setq inhibit-startup-message t
        initial-scratch-message ""
        initial-buffer-choice t)

  ;; Aggiorna l'indice dei pacchetti se è vuoto o non ancora caricato.
  ;; Evita il 404 "package-X.tar: Not found" causato da un indice MELPA scaduto.
  (unless package-archive-contents
    (package-refresh-contents))

  (unless (package-installed-p 'use-package)
    (package-install 'use-package))

  (require 'use-package)
  (setq use-package-always-ensure t)
#+end_src

* Minibuffer & Completamento
Tutto lo stack di ricerca, navigazione e completamento in un unico posto.
=vertico= + =consult= + =embark= coprono la navigazione; =corfu= + =cape= il completamento inline.

** Minibuffer: ricerca e navigazione

#+begin_src emacs-lisp
  (use-package vertico
    :init (vertico-mode 1)
    :custom (vertico-cycle t))

  ;; Persiste la history del minibuffer tra sessioni
  (use-package savehist
    :ensure nil
    :init (savehist-mode 1))

  ;; Annotazioni contestuali accanto ai candidati (tipo file, docstring, ecc.)
  (use-package marginalia
    :init (marginalia-mode 1))

  ;; Matching fuzzy per stile "parole separate da spazi"
  (use-package orderless
    :custom
    (completion-styles '(orderless basic))
    (completion-category-defaults nil)
    (completion-category-overrides '((file (styles partial-completion)))))

  (use-package consult
    :bind (("C-x b" . consult-buffer)
           ("C-s"   . consult-line)
           ("C-c s" . consult-ripgrep))
    :custom
    (isearch-invisible 'open))

  (use-package which-key
    :init (which-key-mode)
    :custom (which-key-idle-delay 0.5))
#+end_src

** Embark — azioni contestuali

=embark= aggiunge un layer di azioni su qualsiasi "cosa" sotto il cursore: candidati
nel minibuffer, file, simboli, URL. =embark-consult= collega le due librerie e abilita
=embark-export= sui risultati di consult (es. ripgrep → buffer grep modificabile con wgrep).

#+begin_src emacs-lisp
  ;; :pin gnu — il build MELPA di embark è attualmente rotto (tar 404).
  ;; Entrambi i pacchetti sono disponibili anche su GNU ELPA, usiamo quella.
  (use-package embark
    :pin gnu
    :bind (("C-."   . embark-act)      ; menu azioni contestuale
           ("C-;"   . embark-dwim)     ; azione "ovvia" senza menu
           ("C-h B" . embark-bindings)) ; mostra tutti i bind del contesto corrente
    :custom
    (embark-help-key "?"))

  (use-package embark-consult
    :pin gnu
    :hook (embark-collect-mode . consult-preview-at-point-mode))

  ;; wgrep: rende modificabile il buffer grep/ripgrep prodotto da embark-export
  ;; Workflow: consult-ripgrep → C-. → E (export) → modifica → C-c C-c (salva tutto)
  (use-package wgrep
    :custom (wgrep-auto-save-buffer t))
#+end_src

** Corfu + Cape — completamento inline

=corfu= gestisce il popup; =cape= fornisce i backend quando LSP non è attivo
(testo, dabbrev, path di file, keyword del linguaggio corrente).

#+begin_src emacs-lisp
  (use-package corfu
    :init (global-corfu-mode)
    :custom
    (corfu-auto t)
    (corfu-quit-no-match t))

  (use-package cape
    :init
    (add-hook 'completion-at-point-functions #'cape-dabbrev) ; parole già nel buffer
    (add-hook 'completion-at-point-functions #'cape-file)    ; path dopo /
    (add-hook 'completion-at-point-functions #'cape-keyword)) ; keyword del major-mode
#+end_src

* Org Mode
Setup per docs-as-code, literate programming ed estetica.

#+begin_src emacs-lisp
  (use-package org
    :custom
    ;; Estetica
    (org-hide-emphasis-markers t)
    (org-ellipsis " ▾")
    (org-startup-indented t)
    (org-return-follows-link t)

    ;; Babel
    (org-confirm-babel-evaluate nil)

    ;; Src block — indentazione e comportamento del buffer di editing
    (org-src-tab-acts-natively t)        ; tab si comporta come nel major mode del linguaggio
    (org-src-preserve-indentation t)     ; non aggiunge indentazione spuria al tangle
    (org-edit-src-content-indentation 0) ; evita il doppio indento nel buffer C-c '
    (org-src-window-setup 'current-window) ; C-c ' apre nella finestra corrente

    ;; Enfasi personalizzata
    (org-emphasis-alist
     '(("*" (:foreground "#ff6c6b" :weight bold))
       ("/" (:foreground "#98be65" :slant italic))
       ("_" (:foreground "#da8548" :underline t))
       ("=" (:background "yellow" :foreground "black"))
       ("~" (:foreground "#51afef"))
       ("+" (:strike-through t :foreground "#5B6268"))))

    :config
    (require 'org-tempo)
    (org-babel-do-load-languages
     'org-babel-load-languages
     '((emacs-lisp . t) (python . t) (C . t) (shell . t))))

  ;; Estetica moderna: bullets, liste, tabelle
  (use-package org-modern
    :hook (org-mode . org-modern-mode)
    :custom
    (org-modern-star '("◉" "○" "◈" "◇" "✳"))
    (org-modern-list '((43 . "➤") (45 . "–") (42 . "•")))
    (org-modern-hide-stars nil)
    (org-modern-table nil))

  ;; Mostra i marker di enfasi quando il cursore è sopra
  (use-package org-appear
    :hook (org-mode . org-appear-mode)
    :custom
    (org-appear-autoemphasis t)
    (org-appear-autolinks t))

  ;; TOC automatico aggiornato al salvataggio
  (use-package toc-org
    :hook (org-mode . toc-org-mode))

  ;; Ambiente di scrittura centrato — margini laterali per leggibilità su schermi larghi
  ;; :commands dichiara olivetti-mode come autoload al byte-compiler,
  ;; eliminando il warning "reference to free variable 'olivetti-mode'"
  (use-package olivetti
    :commands olivetti-mode
    :hook (org-mode . olivetti-mode)
    :custom (olivetti-body-width 100))

  ;; Exporter: GitHub Flavored Markdown e Qiita Markdown
  (use-package ox-gfm)
  (use-package ox-qmd)
#+end_src

* Terminale (Vterm)
Emulatori, tiling e utility di sessione.

#+begin_src emacs-lisp
  (use-package vterm
    :commands vterm
    :custom (vterm-max-scrollback 10000))

  (use-package vterm-toggle
    :bind (("C-c t" . vterm-toggle)
           ("C-c T" . vterm-toggle-cd))
    :custom (vterm-toggle-fullscreen-p nil)
    :config
    (add-to-list 'display-buffer-alist
                 '((lambda (buf-name _)
                     (with-current-buffer buf-name (eq major-mode 'vterm-mode)))
                   (display-buffer-reuse-window display-buffer-at-bottom)
                   (reusable-frames . visible)
                   (window-height . 0.3))))

  (use-package multi-vterm
    :bind ("C-c v c" . multi-vterm))

  (global-set-key (kbd "C-c r") 'rename-buffer)
#+end_src
